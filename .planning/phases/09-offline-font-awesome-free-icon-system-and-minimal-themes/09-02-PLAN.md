    Create falkon/src/lib/agent/promethusfontloader.cpp:

    Include QFontDatabase and qlogging. Implement registerAll() using the null-terminated const char* array pattern from 09-PATTERNS.md:

    ```cpp
    #include "promethusfontloader.h"
    #include <QFontDatabase>

    void PromethusFontLoader::registerAll()
    {
        static const char* const kFontPaths[] = {
            ":/fonts/poppins/Poppins-Thin.ttf",
            ":/fonts/poppins/Poppins-Light.ttf",
            ":/fonts/poppins/Poppins-ExtraBold.ttf",
            ":/fonts/poppins/Poppins-Black.ttf",
            ":/fonts/spacemono/SpaceMono-Regular.ttf",
            nullptr
        };
        for (int i = 0; kFontPaths[i]; ++i) {
            const int id = QFontDatabase::addApplicationFont(
                QLatin1String(kFontPaths[i]));
            if (id == -1) {
                qWarning("PromethusFontLoader: failed to load %s", kFontPaths[i]);
            }
        }
    }
    ```

    Note: "PromethusFontLoader" (single 's') is the canonical spelling per 09-UI-SPEC.md (see "PromethusFontLoader: failed to load" message). Use this spelling throughout — do not add a second 's'.

    ```cpp
    #pragma once
    #include "falkon_defs.h"
    #include <QObject>
    #include <QIcon>
    #include <QHash>
    #include <QPair>
    #include <QString>

    class FALKON_EXPORT PrometheusIconResolver : public QObject
    {
        Q_OBJECT
    public:
        // Central lookup: actionId must be one of the action IDs in 09-UI-SPEC.md.
        // 1. Tries :/icons/fa/{glyph}.svg from Qt resources.
        // 2. If null: tries QIcon::fromTheme(fallback, QIcon()).
        // 3. If both fail: emits qWarning and returns 1x1 transparent QIcon.
        static QIcon icon(const QString &actionId);

        static PrometheusIconResolver* instance();

    private:
        explicit PrometheusIconResolver(QObject* parent = nullptr);

        // value: QPair<resourcePath, themeFallback>
        // themeFallback may be empty (no fallback defined for that action)
        static QHash<QString, QPair<QString, QString>> s_actionMap;
        static void initActionMap();
        static bool s_mapInitialized;
    };
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| actionId string → resource path | Caller-provided actionId is used to index into a static compile-time map; no path traversal possible |
| Resource path → QIcon load | Path is a compile-time constant from the action map; not user-controllable |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-09-03-A | Tampering | Caller passes crafted actionId to navigate resource path | mitigate | All resource paths are compile-time constants in s_actionMap; unknown actionId returns QIcon() not a path-based load; no string interpolation into resource paths |
| T-09-03-B | Information Disclosure | qWarning leaks internal path on missing icon | accept | Warning message is developer-facing only; contains compile-time resource path (not runtime user data); consistent with ICON-04 release gate design |
| T-09-03-C | Denial of Service | icon() called from many UI threads simultaneously | mitigate | s_actionMap is initialized once in initActionMap() called from instance(); Qt singleton pattern via Q_GLOBAL_STATIC is thread-safe for initialization; icon() is read-only after init |
| T-09-03-D | Elevation of Privilege | Remote icon load via QIcon::fromTheme pointing to network | accept | QIcon::fromTheme resolves from local theme directories only on all platforms; no HTTP scheme is resolved |
</threat_model>
